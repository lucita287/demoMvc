drop table ST_PRODUCTO;
drop table ST_CLIENTE;
DROP TABLE ST_CLIENTE_PRODUCTO;
create table ST_PRODUCTO(
  ID_PRODUCTO NUMBER(5),
  PRODUCTO VARCHAR2(100),
  DESCRIPCION VARCHAR2(300),
  PRECIO      NUMBER(12,2),
  EXISTENCIAS NUMBER(8),
  PATH_PRODUCTO VARCHAR2(150),
  PRIMARY KEY(ID_PRODUCTO)
);

create table ST_CLIENTE(
ID_CLIENTE NUMBER(5),
Nombre VARCHAR2(200),
Apellidos VARCHAR2(200),
Direccion VARCHAR2(500),
Email VARCHAR2(150),
Telefono VARCHAR2(50),
Movil VARCHAR2(50),
Fax VARCHAR2(200),
PRIMARY KEY(ID_CLIENTE)
);
/
create table ST_CLIENTE_PRODUCTO(
ID_VENTA NUMBER(8),
ID_CLIENTE NUMBER(5),
ID_PRODUCTO NUMBER(5),
CANTIDAD NUMBER(8),
FECHA_COMPRA DATE DEFAULT SYSDATE,
ID_FORMA_PAGO NUMBER(8),
DESCUENTO NUMBER(12,2),
PRECIO NUMBER(12,2),
PRIMARY KEY(ID_CLIENTE,ID_PRODUCTO,ID_VENTA),
 CONSTRAINT fk_cliente_venta
    FOREIGN KEY (ID_CLIENTE)
    REFERENCES ST_CLIENTE(ID_CLIENTE),
 CONSTRAINT fk_producto_venta
    FOREIGN KEY (ID_PRODUCTO)
    REFERENCES ST_PRODUCTO(ID_PRODUCTO)   
);
